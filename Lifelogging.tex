\begin{slide}

    \begin{itemize}
        \item Hashtags, e.g. \textbf{\#alcohol}, \textbf{\#sleep}, \textbf{\#tv}
        \item Amounts, e.g. \textbf{Double Espresso \#caffeine 150mg \#drugs}
